  <script>
    function isMobileDevice() {
      return window.innerWidth <= 768;
    }

    window.onload = () => {
      const cookie = document.getElementById('cookie');
      const leftHalf = document.querySelector('.left');
      const rightHalf = document.querySelector('.right');
      const cookieSound = document.getElementById('cookieSound');

      let broken = false;
      const strength = 30;
      let clickCount = 0;


      cookie.addEventListener('click', () => {
  clickCount++;

  // CLICK 1 → CRACK
  if (clickCount === 1) {
    cookie.classList.add('crack');
  }

  // CLICK 2 → BREAK + SHOW TEXT BETWEEN HALVES
  if (clickCount === 2) {
    broken = true;
    cookie.classList.remove('crack');
    cookie.classList.add('break');

    cookieSound?.play().catch(() => {});

    document.getElementById('text').classList.add('show');
  }

  // CLICK 3 → FADE HALVES AWAY
  if (clickCount === 3) {
    gsap.to(cookie, {
      opacity: 0,
      scale: 0.85,
      duration: 1,
      ease: "power2.inOut",
      onComplete: () => {
        cookie.classList.add('hide-cookie');
      }
    });
  }
});



      if (!isMobileDevice()) {
        cookie.addEventListener('mousemove', (e) => {
          const rect = cookie.getBoundingClientRect();
  const x = e.clientX - rect.left - rect.width / 2;
  const y = e.clientY - rect.top - rect.height / 2;

  if (!broken) {
    gsap.to(cookie, {
      x: x / strength,
      y: y / strength,
      rotationY: x / 18,
      rotationX: -y / 18,
      duration: 0.4,
      ease: "power3.out"
    });
  } else {
    gsap.to(leftHalf, {
      x: (-200 + x / 20),
      y: y / 25,
      rotationZ: -x / 40,
      duration: 0.4,
      ease: "power3.out"
    });

    gsap.to(rightHalf, {
      x: (200 + x / 20),
      y: y / 25,
      rotationZ: x / 40,
      duration: 0.4,
      ease: "power3.out"
            });
          }
        });

        cookie.addEventListener('mouseleave', () => {
          if (!broken) {
            gsap.to(cookie, {
              x: 0, y: 0,
              rotationX: 0, rotationY: 0,
              duration: 0.6,
              ease: "power3.out"
            });
          }
        });
      }

      /* AUTO BREAK */
